    <script>

        // --- PENTING: GANTIKAN URL INI DENGAN URL WEB APP ANDA ---

        const WEB_APP_URL = 'URL_WEB_APP_ANDA_DI_SINI'; 


        // Logic Preview Gambar

        const fileInput = document.getElementById('mediaFiles');

        const imgPreview1 = document.getElementById('imgPreview1');

        const imgPreview2 = document.getElementById('imgPreview2');

        const placeholders = document.querySelectorAll('.preview-placeholder');


        fileInput.addEventListener('change', function(e) {

            const files = e.target.files;

            

            // Reset

            imgPreview1.style.display = 'none';

            imgPreview2.style.display = 'none';

            placeholders.forEach(p => p.style.display = 'block');


            if (files.length > 2) {

                alert("Maksimum 2 gambar sahaja dibenarkan.");

                fileInput.value = "";

                return;

            }


            // Loop untuk preview

            Array.from(files).forEach((file, index) => {

                if (file.type.startsWith('image/')) {

                    const reader = new FileReader();

                    reader.onload = function(e) {

                        if (index === 0) {

                            imgPreview1.src = e.target.result;

                            imgPreview1.style.display = 'block';

                            placeholders[0].style.display = 'none';

                        } else if (index === 1) {

                            imgPreview2.src = e.target.result;

                            imgPreview2.style.display = 'block';

                            placeholders[1].style.display = 'none';

                        }

                    }

                    reader.readAsDataURL(file);

                }

            });

        });


        // Logic Hantar Borang

        document.getElementById('laporanForm').addEventListener('submit', function(e) {

            e.preventDefault();


            if (WEB_APP_URL.includes('URL_WEB_APP')) {

                alert('Sila masukkan URL Web App Google Script dalam kod HTML!');

                return;

            }


            document.getElementById('loadingScreen').style.display = 'flex';


            const nama = document.getElementById('nama').value;

            const tempat = document.getElementById('tempat').value;

            const ringkasan = document.getElementById('ringkasan').value;

            const files = document.getElementById('mediaFiles').files;


            // Fungsi baca fail

            const readFile = (file) => {

                return new Promise((resolve, reject) => {

                    const reader = new FileReader();

                    reader.onload = () => resolve({

                        base64: reader.result,

                        type: file.type,

                        name: file.name

                    });

                    reader.onerror = reject;

                    reader.readAsDataURL(file);

                });

            };


            const filePromises = Array.from(files).map(readFile);


            Promise.all(filePromises).then(fileData => {

                const payload = {

                    nama: nama,

                    tempat: tempat,

                    ringkasan: ringkasan,

                    files: fileData

                };


                fetch(WEB_APP_URL, {

                    method: 'POST',

                    body: JSON.stringify(payload)

                })

                .then(res => res.json())

                .then(data => {

                    document.getElementById('loadingScreen').style.display = 'none';

                    if(data.result === 'success') {

                        alert('Laporan Berjaya Dihantar!');

                        // Reset borang

                        document.getElementById('laporanForm').reset();

                        imgPreview1.style.display = 'none';

                        imgPreview2.style.display = 'none';

                        placeholders.forEach(p => p.style.display = 'block');

                    } else {

                        alert('Ralat: ' + JSON.stringify(data));

                    }

                })

                .catch(err => {

                    document.getElementById('loadingScreen').style.display = 'none';

                    alert('Gagal menyambung ke server.');

                    console.error(err);

                });

            });

        });
